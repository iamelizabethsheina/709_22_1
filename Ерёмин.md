(

(

(
    provider_id uuid references provider_nfd,
    product_id uuid references product_vcw,
    price numeric,
    quantity int,
    primary key (product_id, provider_id)
);

create table vehicle_usk
(
    id                uuid primary key,
    model             text,
    body_volume       int,
    lifting_capacity  int
);

create table provider_vehicle_info_aat
(
    provider_id uuid references provider_nfd,
    vehicle_id uuid references vehicle_usk,

    primary key (vehicle_id, provider_id)
);
