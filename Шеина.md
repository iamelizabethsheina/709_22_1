--Вывести минимальную и максимальную цену,
-- средний размер экрана и суммарную стоимость (price * quantity) электронных книг бренда Amazon.
select min(price), max(price), avg(screen_size), sum(price * quantity) from supplier_product_info_jkc spiun
join product_xww prod on spiun.product_id = prod.id
join supplier_tdz sup on spiun.supplier_id = sup.id
where brand = 'Amazon'
and category = 'Электронные книги'

--Вывести модели электронных книг с наименьшим весом
-- и наименьшей ценой среди электронных книг с наименьшим весом.
select model from product_xww prod
join supplier_product_info_jkc spiun on spiun.product_id = prod.id
where screen_size = (
    select min(screen_size) from product_xww prod
    join supplier_product_info_jkc spiun on spiun.product_id = prod.id
    where screen_size > 0
)
and screen_size > 0
and weight = (
    select min(weight) from product_xww prod
    join supplier_product_info_jkc spiun on spiun.product_id = prod.id
    where screen_size > 0
)
and screen_size > 0
